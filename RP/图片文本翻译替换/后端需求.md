# 图片文本翻译替换 - 后端需求文档（V1.0）

**定位：** NextJS API Route，无数据库、无 ORM、无 Redis、无队列。接收图片 → 调用魔塔 API → 处理图片 → 返回结果，一次请求完成全部工作。

---

## 一、目录结构

```
app/api/translate/route.ts    # 唯一的服务端接口
lib/
├── modelscope.ts             # 魔塔客户端初始化
├── ocr.ts                    # 调用 qwen-vl-plus 识别图片中文
├── translator.ts             # 调用 qwen-turbo 翻译文本
└── image-processor.ts        # Sharp + Canvas 替换图片文字
```

### 4.2 图片文字识别（lib/ocr.ts）

调用 `qwen-vl-plus` 识别图片中的中文文本及位置。

```typescript
import { client } from './modelscope';

export interface OcrItem {
  text: string;
  x: number;
  y: number;
  width: number;
  height: number;
  fontSize: number;
  color: string;    // 文字颜色，如 #000000
  bgColor: string;  // 背景颜色，如 #ffffff
}

export async function recognizeText(base64: string, format = 'jpeg'): Promise<OcrItem[]> {
  const mime = format === 'png' ? 'image/png' : format === 'webp' ? 'image/webp' : 'image/jpeg';

  const res = await client.chat.completions.create({
    model: 'qwen-vl-plus',
    messages: [{
      role: 'user',
      content: [
        {
          type: 'image_url',
          image_url: { url: `data:${mime};base64,${base64}` }
        },
        {
          type: 'text',
          text: `识别图片中所有中文文本，返回位置和样式信息。
仅返回JSON数组，不要其他内容：
[{"text":"文字","x":左上角x,"y":左上角y,"width":宽,"height":高,"fontSize":字号,"color":"#颜色","bgColor":"#背景色"}]
没有中文则返回 []`
        }
      ]
    }],
    temperature: 0.1,
    max_tokens: 2048
  });

  const content = res.choices[0]?.message?.content || '[]';
  const match = content.match(/\[[\s\S]*\]/);
  if (!match) return [];

  try {
    return (JSON.parse(match[0]) as any[]).filter(i => i.text && typeof i.x === 'number');
  } catch {
    return [];
  }
}
```

### 4.3 文字翻译（lib/translator.ts）

调用 `qwen-turbo` 批量翻译文本。

```typescript
import { client } from './modelscope';

const LANG_MAP: Record<string, string> = {
  en: '英语', ja: '日语', ko: '韩语', es: '西班牙语',
  fr: '法语', de: '德语', ru: '俄语', th: '泰语',
  vi: '越南语', id: '印度尼西亚语'
};

export async function translateTexts(texts: string[], targetLang: string): Promise<string[]> {
  if (!texts.length) return [];

  const res = await client.chat.completions.create({
    model: 'qwen-turbo',
    messages: [
      {
        role: 'system',
        content: `将中文翻译成${LANG_MAP[targetLang] || '英语'}，保持简洁，适合图片文字。`
      },
      {
        role: 'user',
        content: `逐行翻译，只返回译文，保持顺序：\n${texts.join('\n')}`
      }
    ],
    temperature: 0.1,
    max_tokens: 1024
  });

  const lines = (res.choices[0]?.message?.content || '')
    .split('\n').map(l => l.trim()).filter(Boolean);

  return texts.map((t, i) => lines[i] || t);
}
```

### 4.4 图片文字替换（lib/image-processor.ts）

用 Canvas 覆盖原文字区域，绘制翻译后的文字。

```typescript
import sharp from 'sharp';
import { createCanvas, loadImage } from 'canvas';

export interface TextReplacement {
  translated: string;
  x: number;
  y: number;
  width: number;
  height: number;
  fontSize: number;
  color: string;
  bgColor: string;
}

export async function replaceText(imageBuffer: Buffer, items: TextReplacement[]): Promise<Buffer> {
  const { width = 800, height = 600 } = await sharp(imageBuffer).metadata();

  // 转为 PNG 供 Canvas 使用
  const pngBuffer = await sharp(imageBuffer).png().toBuffer();
  const image = await loadImage(pngBuffer);

  const canvas = createCanvas(width, height);
  const ctx = canvas.getContext('2d');
  ctx.drawImage(image, 0, 0);

  for (const item of items) {
    const { translated, x, y, width: w, height: h, fontSize, color, bgColor } = item;

    // 用背景色覆盖原文字
    ctx.fillStyle = bgColor;
    ctx.fillRect(x - 2, y - 2, w + 4, h + 4);

    // 计算合适字号（防止溢出）
    let fs = fontSize;
    while (fs > 8) {
      ctx.font = `${fs}px sans-serif`;
      if (ctx.measureText(translated).width <= w) break;
      fs -= 1;
    }

    // 绘制译文
    ctx.fillStyle = color;
    ctx.font = `${fs}px sans-serif`;
    ctx.textBaseline = 'top';
    ctx.fillText(translated, x, y + (h - fs) / 2);
  }

  return sharp(canvas.toBuffer('image/png')).jpeg({ quality: 90 }).toBuffer();
}
```

### 4.5 API Route（app/api/translate/route.ts）

整合以上三步，对外暴露唯一接口。

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { recognizeText } from '@/lib/ocr';
import { translateTexts } from '@/lib/translator';
import { replaceText } from '@/lib/image-processor';

export async function POST(request: NextRequest) {
  try {
    const { id, base64, format = 'jpeg', targetLang } = await request.json();

    if (!id || !base64 || !targetLang) {
      return NextResponse.json({ id, success: false, error: '缺少必要参数' }, { status: 400 });
    }

    // Step 1: OCR 识别
    const ocrItems = await recognizeText(base64, format);
    if (!ocrItems.length) {
      return NextResponse.json({ id, success: false, error: '未识别到中文文本' });
    }

    // Step 2: 翻译
    const translated = await translateTexts(ocrItems.map(i => i.text), targetLang);

    // Step 3: 图片处理
    const imageBuffer = Buffer.from(base64, 'base64');
    const resultBuffer = await replaceText(
      imageBuffer,
      ocrItems.map((item, i) => ({ ...item, translated: translated[i] }))
    );

    return NextResponse.json({
      id,
      success: true,
      translatedBase64: resultBuffer.toString('base64'),
      texts: ocrItems.map((item, i) => ({
        original: item.text,
        translated: translated[i]
      }))
    });

  } catch (err: any) {
    console.error('[translate]', err);
    return NextResponse.json({ success: false, error: err.message || '处理失败' }, { status: 500 });
  }
}
```

## 五、next.config.js 配置

`sharp` 和 `canvas` 是 Node.js 原生模块，需要告知 NextJS 不要打包它们：

```javascript
// next.config.js
/** @type {import('next').NextConfig} */
const nextConfig = {
  serverExternalPackages: ['sharp', 'canvas']
};

module.exports = nextConfig;
```
