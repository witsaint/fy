## 四、类型定义

```typescript
// types/index.ts

export type ImageStatus = 'idle' | 'processing' | 'done' | 'error';

export interface ImageItem {
  id: string;                   // crypto.randomUUID()
  file: File;                   // 原始文件
  previewUrl: string;           // URL.createObjectURL(file)
  status: ImageStatus;
  translatedBase64?: string;    // 服务端返回的处理结果
  texts?: { original: string; translated: string }[];  // 识别到的文本
  errorMessage?: string;
}
```

## 五、核心 Hook

```typescript
// hooks/useTranslate.ts
'use client';
import { useState, useCallback } from 'react';
import { ImageItem } from '@/types';

export function useTranslate() {
  const [images, setImages] = useState<ImageItem[]>([]);
  const [targetLang, setTargetLang] = useState('en');
  const [isProcessing, setIsProcessing] = useState(false);

  const updateImage = useCallback((id: string, updates: Partial<ImageItem>) => {
    setImages(prev => prev.map(img => img.id === id ? { ...img, ...updates } : img));
  }, []);

  const addImages = useCallback((files: File[]) => {
    const newItems: ImageItem[] = files.map(file => ({
      id: crypto.randomUUID(),
      file,
      previewUrl: URL.createObjectURL(file),
      status: 'idle'
    }));
    setImages(prev => [...prev, ...newItems]);
  }, []);

  const removeImage = useCallback((id: string) => {
    setImages(prev => {
      const img = prev.find(i => i.id === id);
      if (img) URL.revokeObjectURL(img.previewUrl);
      return prev.filter(i => i.id !== id);
    });
  }, []);

  const clearAll = useCallback(() => {
    setImages(prev => { prev.forEach(i => URL.revokeObjectURL(i.previewUrl)); return []; });
  }, []);

  // 处理单张图片
  const processOne = useCallback(async (image: ImageItem) => {
    updateImage(image.id, { status: 'processing' });
    try {
      // 文件转 base64（去掉 data:image/xxx;base64, 前缀）
      const base64 = await new Promise<string>((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = () => resolve((reader.result as string).split(',')[1]);
        reader.onerror = reject;
        reader.readAsDataURL(image.file);
      });

      const res = await fetch('/api/translate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          id: image.id,
          base64,
          format: image.file.type.split('/')[1],
          targetLang
        })
      });

      const data = await res.json();

      if (data.success) {
        updateImage(image.id, {
          status: 'done',
          translatedBase64: data.translatedBase64,
          texts: data.texts
        });
      } else {
        updateImage(image.id, { status: 'error', errorMessage: data.error });
      }
    } catch {
      updateImage(image.id, { status: 'error', errorMessage: '网络错误，请重试' });
    }
  }, [targetLang, updateImage]);

  // 批量处理：前端逐张调用同一个 /api/translate 接口，每批最多3张并发
  // 例如上传了10张：第1批同时处理1/2/3，完成后第2批处理4/5/6，以此类推
  const startTranslate = useCallback(async () => {
    const pending = images.filter(img => img.status === 'idle' || img.status === 'error');
    if (!pending.length) return;
    setIsProcessing(true);
    const CONCURRENCY = 3;
    for (let i = 0; i < pending.length; i += CONCURRENCY) {
      await Promise.all(pending.slice(i, i + CONCURRENCY).map(processOne));
    }
    setIsProcessing(false);
  }, [images, processOne]);

  return { images, targetLang, isProcessing, setTargetLang, addImages, removeImage, clearAll, startTranslate };
}
```

## 六、组件说明

### UploadZone.tsx

- 使用 `react-dropzone`
- 接受 JPG/PNG/WebP，单张≤10MB，最多20张
- 超出数量/格式/大小时 toast 提示
- 拖拽时边框高亮

### ImageCard.tsx

每张图片显示：
- **待处理**：原图缩略图，灰色边框
- **处理中**：原图缩略图 + 加载动画，蓝色边框
- **已完成**：翻译后图片缩略图，绿色边框 + 下载按钮
- **失败**：原图缩略图，红色边框 + 错误提示 + 重试按钮

单张下载：
```typescript
const link = document.createElement('a');
const ext = image.file.name.split('.').pop() || 'jpg';
link.href = `data:image/${ext};base64,${image.translatedBase64}`;
link.download = image.file.name.replace(/\.[^.]+$/, '') + `_translated.${ext}`;
link.click();
```

### LanguageSelector.tsx

下拉选择目标语言，支持10种语言：
`en / ja / ko / es / fr / de / ru / th / vi / id`

### 底部操作栏（内联在 page.tsx）

- **开始翻译**：有待处理图片时可点击
- **批量下载**：有已完成图片时可点击，用 JSZip 打包
- **清空**：清空全部图片

批量下载：
```typescript
import JSZip from 'jszip';
import { saveAs } from 'file-saver';

const zip = new JSZip();
doneImages.forEach(img => {
  const ext = img.file.name.split('.').pop() || 'jpg';
  const name = img.file.name.replace(/\.[^.]+$/, '') + `_translated.${ext}`;
  zip.file(name, img.translatedBase64!, { base64: true });
});
const blob = await zip.generateAsync({ type: 'blob' });
saveAs(blob, `translated_${Date.now()}.zip`);
```
